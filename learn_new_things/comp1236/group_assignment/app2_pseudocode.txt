# Program 2

# This app asks user for input: a, b, amount_to_print, should_continue
# It outputs sequence that go by Un = a*n + b, sum_of_terms
# n is a positive integer, increments each count
# It asks whether to repeat the program

Start
While true:
	Input a, b, amount_to_print
	Set n = 1, sum_of_terms = 0
	
	For n <= amount_to_print:
		Set term = a*n + b
		Set sum_of_terms += term
		Output term, sum_of_terms
		Set n += 1
	Endfor
	
	Input should_continue
	If should_continue == 'yes':
		Then pass
	Else:
		break
Endwhile
Stop
